fold_data <- vfold_cv(train_data, v = 5, repeats = 5, strata = BodyTemp)

#Create a recipe for the data and fitting. 
data_recipe <- recipe(BodyTemp ~ ., data = train_data) %>%
  step_dummy(all_nominal(), -all_outcomes())
